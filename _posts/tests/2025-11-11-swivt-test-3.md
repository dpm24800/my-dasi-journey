#### 29.10. Read all product sales data and show it using the stack plot.
```py
import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv('company_sales_data.csv')

total_facecream = df['facecream'].sum()
total_facewash = df['facewash'].sum()
total_toothpaste = df['toothpaste'].sum()
total_bathingsoap = df['bathingsoap'].sum()
total_shampoo = df['shampoo'].sum()
total_moisturizer = df['moisturizer'].sum()

print("total_facecream:", total_facecream)
print("total_facewash:", total_facewash)
print("total_toothpaste:", total_toothpaste)
print("total_bathingsoap:", total_bathingsoap)
print("total_shampoo:", total_shampoo)
print("total_moisturizer:", total_moisturizer)

plt.pie([total_facecream, total_facewash, total_toothpaste, total_bathingsoap, total_shampoo, total_moisturizer], labels = ['facecream', 'facewash', 'toothpaste', 'bathingsoap', 'shampoo', 'moisturizer'], colors=["red", "hotpink", "g", "#93B41B", 'tomato', "#5271C0"], explode=[0, 0, 0, 0, 0.03, 0], startangle=0, shadow = False)
plt.show()
```
